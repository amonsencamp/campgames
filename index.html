<h1>THROW A SANDWICH AT JEFF</h1>

<p>HOLD SPACE TO AIM • RELEASE TO THROW</p>


<canvas id="game" width="600" height="300"></canvas>


<script>

const canvas = document.getElementById("game");

const ctx = canvas.getContext("2d");


// Game state

let angle = 45;

let aimingUp = true;

let isHolding = false;

let sandwich = null;

let gravity = 0.3;

let score = 0;

let shots = 5;

let gameOver = false;

let initials = "";


// Jeff

const jeff = {

  x: 500,

  y: 220,

  width: 20,

  height: 40,

  happyTimer: 0

};


// Input

document.addEventListener("keydown", e => {

  if (e.code === "Space") isHolding = true;


  if (gameOver && initials.length < 3) {

    if (/^[a-zA-Z]$/.test(e.key)) {

      initials += e.key.toUpperCase();

    }

  }

});


document.addEventListener("keyup", e => {

  if (e.code === "Space" && !gameOver) {

    isHolding = false;

    if (!sandwich && shots > 0) launchSandwich();

  }

});


function launchSandwich() {

  let power = 8;

  let rad = angle * Math.PI / 180;


  sandwich = {

    x: 100,

    y: 220,

    vx: Math.cos(rad) * power,

    vy: -Math.sin(rad) * power,

    alive: true

  };


  shots--;

}


// Game loop

function update() {

  if (!gameOver) {

    // Aim oscillation

    if (isHolding) {

      if (aimingUp) angle += 1.5;

      else angle -= 1.5;


      if (angle >= 90) aimingUp = false;

      if (angle <= 0) aimingUp = true;

    }


    // Sandwich physics

    if (sandwich) {

      sandwich.x += sandwich.vx;

      sandwich.y += sandwich.vy;

      sandwich.vy += gravity;


      // Hit Jeff

      if (

        sandwich.x > jeff.x &&

        sandwich.x < jeff.x + jeff.width &&

        sandwich.y > jeff.y &&

        sandwich.y < jeff.y + jeff.height

      ) {

        score++;

        jeff.happyTimer = 30;

        sandwich = null;

      }


      // Ground hit

      if (sandwich && sandwich.y > 260) {

        sandwich = null;

      }

    }


    if (shots === 0 && !sandwich) {

      gameOver = true;

    }

  }


  if (jeff.happyTimer > 0) jeff.happyTimer--;


  draw();

  requestAnimationFrame(update);

}


// Draw everything

function draw() {

  ctx.fillStyle = "black";

  ctx.fillRect(0, 0, canvas.width, canvas.height);


  // Ground

  ctx.fillStyle = "#003300";

  ctx.fillRect(0, 260, 600, 40);


  // Catapult

  ctx.fillStyle = "#00ff88";

  ctx.fillRect(80, 230, 40, 10);


  // Arm

  let armLength = 40;

  let rad = angle * Math.PI / 180;

  let x2 = 100 + Math.cos(rad) * armLength;

  let y2 = 230 - Math.sin(rad) * armLength;


  ctx.beginPath();

  ctx.moveTo(100, 230);

  ctx.lineTo(x2, y2);

  ctx.strokeStyle = "#00ff88";

  ctx.stroke();


  // Sandwich

  if (sandwich) {

    ctx.fillStyle = "#ffcc66";

    ctx.fillRect(sandwich.x, sandwich.y, 10, 6);

  }


  // Jeff

  ctx.fillStyle = "#00ff88";

  ctx.fillRect(jeff.x, jeff.y, jeff.width, jeff.height);


  // Face

  ctx.fillStyle = "black";

  if (jeff.happyTimer > 0) {

    ctx.fillText(":)", jeff.x, jeff.y - 5);

  } else {

    ctx.fillText(":|", jeff.x, jeff.y - 5);

  }


  // UI

  ctx.fillStyle = "#00ff88";

  ctx.fillText("SCORE: " + score, 10, 20);

  ctx.fillText("SHOTS: " + shots, 10, 40);


  if (gameOver) {

    ctx.fillText("GAME OVER", 250, 120);

    ctx.fillText("ENTER INITIALS: " + initials, 200, 150);

  }

}


update();

</script>
